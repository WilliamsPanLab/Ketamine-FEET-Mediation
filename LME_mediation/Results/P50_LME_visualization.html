# combine all CADSS and 5D-ASC subcomponents of all three visits
non_brain_data_post_pre <- Reduce(function(x, y) merge(x, y, by = c("Subjects", "Dosage"),
    all = TRUE), list(data_repeated_post_pre, data_once_pervisit))

data_fortest <- non_brain_data_post_pre %>%
    select(Subjects, Dosage, CADSS_depersonalization_post_pre, CADSS_derealization_post_pre,
        CADSS_amnesia_post_pre, dascscore_anxiety, dascscore_bliss, dascscore_impair) %>%
    pivot_longer(-c(Subjects, Dosage), names_to = "variables", values_to = "value") %>%
    mutate(Dosage = factor(Dosage, levels = dosage_level))

# by subject by variable
plot_variables <- unique(data_fortest$variables)

plot_titles <- unique(data_fortest$variables)
figure_width = 0.7
figure_height = 1.2
boxplot_width = 0.4/figure_width
colors_network = c("#E5BBD8", "#E5BBD8", "#ffc100", "#E5BBD8", "#ffc100", "#ffc100",
    "#ffc100")
xlabel_text = c("Placebo", "0.05mg/kg", "0.5mg/kg")
# xlabel_text = c('Placebo','0.5mg/kg')
for (ivar in 1:length(plot_variables)) {
    # summarize mean and sd
    data_fortest_sta <- data_fortest %>%
        filter(variables == plot_variables[ivar]) %>%
        group_by(Dosage) %>%
        summarise(count = sum(!is.na(value)), sd = sd(value, na.rm = TRUE), se = sd/sqrt(count),
            avg = mean(value, na.rm = TRUE))

    ylim_max <- max(data_fortest_sta$avg) + 1.2 * max(data_fortest_sta$se)
    ylim_min <- min(data_fortest_sta$avg) - 1.2 * max(data_fortest_sta$se)

    if (ylim_min < 0 & min(data_fortest_sta$avg) >= 0) {
        ylim_min = 0

    }

    colors_network_plot <- c(colors_network[ivar], colors_network[ivar], colors_network[ivar])
    myplot <- data_fortest_sta %>%
        ggplot(aes(x = Dosage, y = avg)) + geom_col(aes(group = Dosage, fill = Dosage),
        alpha = 1, size = 0.25) + geom_errorbar(aes(group = Dosage, color = Dosage,
        ymin = avg - se, ymax = avg + se), width = 0, size = 0.25) + scale_x_discrete(breaks = waiver(),
        labels = xlabel_text) + theme_classic() + theme(axis.ticks = element_line(size = 0.2),
        axis.line = element_line(size = 0.2), axis.text.x = element_text(size = 9),
        axis.title.y = element_blank(), strip.text.x = element_text(size = 8, colour = "black"),
        panel.margin = unit(0.25, "lines"), legend.position = "bottom", legend.margin = margin(t = -10),
        legend.spacing.y = unit(-0.05, "cm")) + scale_colour_manual(values = colors_network_plot) +
        scale_fill_manual(values = colors_network_plot) + guides(color = guide_legend(override.aes = list(alpha = 0.5))) +
        guides(col = guide_legend(nrow = 1)) + ggtitle(plot_variables[ivar])

    if (ylim_min <= 0) {
        myplot <- myplot + geom_hline(yintercept = 0, size = 0.1)
    }

    figure_width = 1.3
    xtitle_size = 6
    title_size = 6
    legend_title_size = 3
    myplot <- add_theme(myplot, xtitle_size, title_size, legend_title_size)
    myplot <- myplot + scale_colour_manual(values = colors_network_plot) + scale_fill_manual(values = colors_network_plot)
    leg <- get_legend(myplot)
    plot_leg <- as_ggplot(leg)
    myplot <- myplot + theme(legend.position = "off")
    print(myplot)
}
